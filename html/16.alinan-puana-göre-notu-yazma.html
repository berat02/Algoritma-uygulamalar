Öğrencinin sınavdan almış olduğu
puana göre notunu ekrana yazdıran programın kodlarını yazınız
Algoritma
1) Başla
2) sayı gir
3) sayı > 0 ve sayı < 50 ise
4) notu( 1 )
5) sayı >= 50 ve sayı < 60 ise
6) notu( 2 )
7) sayı >= 60 ve sayı < 70 ise
8) notu( 3 )
9) sayı >= 70 ve sayı < 85 ise
10) notu( 4 )
11) sayı >= 85 ve sayı <= 100 ise
12) notu( 5 )
13) Bitir
Java Script

function notu_bul() {
var sayi = Number(document.getElementById("sayi").value);
var sonuc = document.getElementById("sonuc");
if (sayi > 0 && sayi < 50) {
sonuc.innerHTML = "Notun: 1";
}else if (sayi >=50 && sayi < 60) {
sonuc.innerHTML = "Notun: 2";
}else if (sayi >=60 && sayi < 70) {
sonuc.innerHTML = "Notun: 3";
}else if (sayi >=70 && sayi < 85) {
sonuc.innerHTML = "Notun: 4";
}else if (sayi >=85 && sayi <=100) {
sonuc.innerHTML = "Notun: 5";
}else{
sonuc.innerHTML = "Notu Düzgün Giriniz";
}
}
