Birbirinden farklı olarak verilen iki adet sayıdan,
büyük olanı bulup gösteren java script ve algoritma'sını tasarlama.
Algoritma
1) Başla
2) sayı1 Giriniz
3) sayı2 Giriniz
4) sayı1 > sayı2
5) sayı1 büyüktür yaz
6) sayı2 > sayı1
7) sayı2 büyüktür yaz
8) Bitir
Java Script

function buyuk_olani_bul(){
var sayi1 = Number(document.getElementById('sayi1').value);
var sayi2 = Number(document.getElementById('sayi2').value);
var sonuc = document.getElementById('sonuc');
if(sayi1 > sayi2){
sonuc.value =("Sayı1 :) "+ sayi1 +" Daha Büyük");
}else if(sayi1 < sayi2){
sonuc.value =("Sayı2 :) "+ sayi2 +" Daha Büyük");
}else{
sonuc.value =("Eşit");
}
}
