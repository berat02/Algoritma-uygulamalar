Girilen vize ve final notlarına göre öğrencinin dersten geçip geçmediğini bulan java script ve algoritma'sını tasarlama.
Algoritma
1) Başla
2) Vize notu Gir
3) Final notu Gir
4) Ortalama = vize * 0.40 + final * 0.60
5) Eğer ortalama >= 60 ise
6) Öğrenci geçti
7) Değilse
8) Öğrenci geçemedi
9) Bitir
Java Script

function vize_final_hesapla(){
var vize = Number(document.getElementById('vize').value);
var final = Number(document.getElementById('final').value);
var sonuc = document.getElementById('sonuc');
var ortalama = vize * 0.40 + final * 0.60;
if(ortalama >= 60){
sonuc.value = ("Ort: "+ortalama+" Öğrenci Geçti");
sonuc.style.color = "green";
}else{
sonuc.value = ("Ort: "+ortalama+" Öğrenci Geçemedi");
sonuc.style.color = "red";
}
}
